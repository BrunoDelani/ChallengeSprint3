 - cors
 - eslint
 - express
 - joi
 - moment
 - mongoose
 - nodemon
 - sucrase
 - typescript

# Database
Cloud MongoDB -> User and database public

# Scripts
Run aplication:
 - npm run dev
Lint aplication:
 - npm run lint:fix
Create dist:
 - npm run build

# Docker
Run aplication with docker:
 - docker-compose up -d
 - docker-compose down
 - docker-compose logs -f api
